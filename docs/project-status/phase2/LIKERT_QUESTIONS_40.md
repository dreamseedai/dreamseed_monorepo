def convert_response_to_score(response: int, reverse: bool) -> int:
    """
    Convert 1-5 Likert response to -2 to +2 score.
    
    Args:
        response: 1 (ì „í˜€ ì•„ë‹ˆë‹¤) to 5 (ë§¤ìš° ê·¸ë ‡ë‹¤)
        reverse: True if reverse-scored item
    
    Returns:
        Score in range [-2, +2]
    """
    if reverse:
        response = 6 - response  # Invert: 1â†’5, 2â†’4, 3â†’3, 4â†’2, 5â†’1
    
    return response - 3  # Convert to -2, -1, 0, +1, +2


def calculate_aptitude_profile(responses: Dict[int, int]) -> Dict[str, float]:
    """
    Calculate dimension scores from 40 responses.
    
    Args:
        responses: {question_id: response_value (1-5)}
    
    Returns:
        {"STEM": 0.45, "Humanities": -0.20, "Artistic": 0.80, "Practical": 0.30}
    """
    dimension_scores = {"STEM": [], "Humanities": [], "Artistic": [], "Practical": []}
    
    for q in QUESTION_METADATA:
        qid = q["id"]
        response = responses.get(qid)
        
        if response is None:
            continue  # Skip missing responses
        
        score = convert_response_to_score(response, q["reverse"])
        dimension_scores[q["dimension"]].append(score)
    
    # Calculate mean for each dimension
    profile = {}
    for dim, scores in dimension_scores.items():
        if len(scores) > 0:
            mean_score = sum(scores) / len(scores)
            normalized = mean_score / 2.0  # Normalize to [-1.0, +1.0]
            profile[dim] = round(normalized, 2)
        else:
            profile[dim] = 0.0
    
    return profile
